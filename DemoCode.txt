    function galleryPrev(e){e.stopPropagation();const c=state.campaigns.find(x=>x.id===galleryCampaignId);if(!c)return;galleryIndex=(galleryIndex-1+c.images.length)%c.images.length;updateGalleryUI();}
    function galleryNext(e){e.stopPropagation();const c=state.campaigns.find(x=>x.id===galleryCampaignId);if(!c)return;galleryIndex=(galleryIndex+1)%c.images.length;updateGalleryUI();}
    let galleryTouchStartX=0;
    document.addEventListener('touchstart',e=>{const v=document.getElementById('gallery-viewport');if(v&&v.contains(e.target))galleryTouchStartX=e.touches[0].clientX;},{passive:true});
    document.addEventListener('touchend',e=>{const v=document.getElementById('gallery-viewport');if(!v||!v.contains(e.target))return;const d=e.changedTouches[0].clientX-galleryTouchStartX;if(Math.abs(d)>50){if(d<0)galleryNext(e);else galleryPrev(e);}},{passive:true});

      const inputSection=state.walletConnected
        ?`<div class="comment-input-wrap"><img src="https://picsum.photos/seed/${state.walletAddress}/40/40" alt="You" class="comment-avatar" style="margin-top:2px"><textarea id="comment-input" class="form-input" placeholder="Share your thoughts on this proposal..." rows="1" onkeydown="handleCommentKeydown(event,'${c.id}')"></textarea><button class="comment-send" onclick="submitComment('${c.id}')" aria-label="Send comment"><i class="fa-solid fa-paper-plane"></i></button></div>`

      return`<div class="comment-section-divider"></div><div class="flex items-center justify-between mb-4"><h3 class="font-display font-semibold text-base" style="color:var(--fg)"><i class="fa-regular fa-comments" style="margin-right:6px;color:var(--accent)"></i>Discussion <span class="text-sm font-normal" style="color:var(--muted)">(${comments.length})</span></h3></div>${inputSection}<div id="comments-list-${c.id}" style="margin-top:0.75rem">${commentsList}</div>`;
    }

    function handleCommentKeydown(e,cid){if(e.key==='Enter'&&!e.shiftKey){e.preventDefault();submitComment(cid);}}

    function submitComment(cid){
      if(!state.walletConnected){showToast('Please connect your wallet first','info');return;}
      const input=document.getElementById('comment-input');if(!input)return;
      const text=input.value.trim();
      if(!text){showToast('Please write a comment','error');return;}
      if(text.length>500){showToast('Comment too long (max 500 characters)','error');return;}
      const c=state.campaigns.find(x=>x.id===cid);if(!c)return;
      c.comments.push({id:'cm'+Date.now(),author:truncateAddress(state.walletAddress),avatar:`https://picsum.photos/seed/${state.walletAddress.slice(0,10)}/40/40`,text,date:'Just now'});
      const listEl=document.getElementById(`comments-list-${cid}`);
      if(listEl){
        const cm=c.comments[c.comments.length-1];
        const html=`<div class="comment-item" style="animation:toastIn 0.3s ease forwards"><img src="${cm.avatar}" alt="You" class="comment-avatar"><div class="comment-body"><div class="comment-header"><span class="comment-author">${cm.author}</span><span class="comment-time">${cm.date}</span></div><p class="comment-text">${escapeHTML(cm.text)}</p></div></div>`;
        if(c.comments.length===1)listEl.innerHTML=html;else listEl.insertAdjacentHTML('afterbegin',html);
      }
      input.value='';input.style.height='auto';renderCampaigns();showToast('Comment posted!','success');
    }

    function handleCreateCampaign(e){
      e.preventDefault();
      const title=$('#create-title').value.trim(),category=$('#create-category').value,
        goal=parseFloat($('#create-goal').value),duration=parseInt($('#create-duration').value,10),
        shortDesc=$('#create-short').value.trim(),description=$('#create-desc').value.trim();
      let valid=false;
      function te(id,show){const el=$(`#${id}`);if(show)el.classList.add('visible');else el.classList.remove('visible');}
      if(!title||title.length>80){te('err-title',true);}else{te('err-title',false);valid=true;}
      if(!category){te('err-category',true);valid=false;}else te('err-category',false);
      if(!goal||goal<100||isNaN(goal)){te('err-goal',true);valid=false;}else te('err-goal',false);
      if(!duration||duration<1||duration>90||isNaN(duration)){te('err-duration',true);valid=false;}else te('err-duration',false);
      if(!shortDesc||shortDesc.length>120){te('err-short',true);valid=false;}else te('err-short',false);
      if(!description){te('err-desc',true);valid=false;}else te('err-desc',false);
      if(!valid)return;
      const images=createImageList.length>0?[...createImageList]:Array.from({length:3},()=>`https://picsum.photos/seed/${generateId()}/600/400`);
      state.campaigns.unshift({id:generateId(),title,shortDesc,description,creator:truncateAddress(state.walletAddress),creatorAvatar:`https://picsum.photos/seed/${generateId()}/80/80`,category,goal,raised:0,donors:0,daysLeft:duration,images,featured:false,updates:[],comments:[]});
      closeModal('create-modal');renderCampaigns();renderStats();showToast(`Proposal "${title}" submitted successfully!`,'success');
    }
